/**
	synchronisation class. Create on of these whenever a critical section is
	necessary. Then, on entering the critical section do:
	<pre>
		Lock alock ( sync );
	</pre>
	the lock will be released when alock goes out of scope.
<p>
	The sync object should be in a scope outside that of alock. A good
	way to do it is to make sync class member, and then for each
	method that needs to control access, have the above line as the
	first line in the method implementation.

	this is a win32 specific class, but the functionality it uses
	should be available in any thread implementation.
*/
class UTILS_DLL_API Sync
{
public:

	/**
		create the synchronisation object
	*/
	Sync()
	{
		::InitializeCriticalSection ( &criticalSection );
	}

	/**
		lock the synchronisation object
	*/
	void lock()
	{
		::EnterCriticalSection ( &criticalSection );
	}

	/**
		release the synchronisation object
	*/
	void release()
	{
		::LeaveCriticalSection ( &criticalSection );
	}

/*
	bool canEnter ()
	{
		return ::TryEnterCriticalSection ( &criticalSection );
	}
*/
	/**
		delete the synchronisation object
	*/
	virtual ~Sync()
	{
		::DeleteCriticalSection ( &criticalSection );
	}

private:
	/// where the OS-level synchronisation object is stored
	CRITICAL_SECTION criticalSection;
};

/**
	A way to lock and release a Sync object easily.
*/
class Lock
{
public:

	// Default Constructor
	Lock()
		: _sync ( 0 )
	{
	}

	Lock ( Sync & aSync )
		: _sync ( &aSync )
	{
		sync().lock();
	}

	void acquire ( Sync & aSync )
	{
		sync ( aSync );
		sync().lock();
	}

	void release ()
	{
		sync().release();
	}

	// Destructor
	virtual ~Lock()
	{
		sync().release();
	}

	Sync & sync()
	{
		if ( _sync == 0 )
			throw runtime_error ( "Unitialised Sync object in Lock" );
		return *_sync;
	}

	Lock & sync ( Sync & aSync )
	{
		_sync = &aSync;
		return *this;
	}

private:
	Sync * _sync;
};
