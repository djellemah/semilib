#ifndef Cloneable_h
#define Cloneable_h

using namespace std;

#include "SmartPointer.h"

/**
	generic clone class. Simply makes a copy of itself and
	returns a smart pointer to the copy.
*/
template <class ToClone>
class Cloneable
{
public:
	SmartPointer<ToClone> * clone() const
	{
		return new ToClone ( *this );
	}

};

#endif
